from flask import Blueprint, render_template, request, redirect, url_for, flash, session main = Blueprint("main", __name__) # ✅ 登入頁設為首頁 @main.route("/", methods=["GET", "POST"]) def login(): if request.method == "POST": username = request.form["username"] password = request.form["password"] # ⛔️ 這裡是簡化的驗證，建議後續改成查資料庫 if username == "admin" and password == "password": session["user"] = username return redirect(url_for("main.dashboard")) flash("Invalid credentials") return render_template("login.html") # ✅ 儀表板頁面（需登入） @main.route("/dashboard") def dashboard(): if "user" not in session: return redirect(url_for("main.login")) return render_template("dashboard.html") # ✅ 管理面板（保護區） @main.route("/access-panel") def access_panel(): if "user" not in session: return redirect(url_for("main.login")) return render_template("access-panel.html")
